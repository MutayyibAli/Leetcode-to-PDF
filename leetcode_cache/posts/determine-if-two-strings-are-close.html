std::vector freq2(26, 0); for (char ch : word1) { freq1[ch - 'a']++; } for (char ch : word2) { freq2[ch - 'a']++; } for (int i = 0; i < 26; i++) { if ((freq1[i] == 0 && freq2[i] != 0) || (freq1[i] != 0 && freq2[i] == 0)) { return false; } } std::sort(freq1.begin(), freq1.end()); std::sort(freq2.begin(), freq2.end()); for (int i = 0; i < 26; i++) { if (freq1[i] != freq2[i]) { return false; } } return true; } }; ``` ``` Java Language: Java class Solution { public boolean closeStrings(String word1, String word2) { int freq1 = new int[26];
